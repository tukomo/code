Program 3: Skewness, Kurtosis, and Visualization


# 1. Install and load necessary libraries
# 'e1071' library is great for calculating skewness and kurtosis
install.packages("e1071")
# 'ggplot2' is a powerful library for data visualization
install.packages("ggplot2")
library(e1071)
library(ggplot2)
# 2. Create a sample dataset
# A slightly larger dataset for better visualization
data <- c(23, 45, 67, 12, 89, 34, 56, 78, 91, 15, 42, 63,
          55, 60, 75, 82, 30, 48, 70, 85, 95, 25, 38, 52)
# 3. Measures of Shape (Skewness and Kurtosis)
# Skewness: Measures the asymmetry of the data distribution
# A value of 0 means a perfectly symmetrical distribution
# A positive value means a right-skewed distribution (tail on the right)
# A negative value means a left-skewed distribution (tail on the left)
skewness_value <- skewness(data)
cat("Skewness:", skewness_value, "\n")
# Kurtosis: Measures the "tailedness" of the data distribution
# A high kurtosis value means the distribution has fatter tails and a sharper peak
# A low kurtosis value means the distribution has lighter tails and a flatter peak
kurtosis_value <- kurtosis(data)
cat("Kurtosis:", kurtosis_value, "\n")
# 4. Data Visualization
# Histogram: Visualizes the distribution of the data
hist(data,
main = "Histogram of Sample Data",
xlab = "Values",
     ylab = "Frequency",
col = "lightblue",
     border = "black")
# The 'hist()' function automatically bins the data and plots the frequencies.
# Boxplot: Provides a visual summary of the five-number summary (min, Q1, median, Q3, max)
# It's excellent for identifying outliers
boxplot(data,
main = "Boxplot of Sample Data",
ylab = "Values",
        col = "lightgreen",
        border = "black")
# Using ggplot2 for more professional-looking plots
# Histogram
ggplot(data.frame(data), aes(x = data)) +
  geom_histogram(binwidth = 10, fill = "salmon", color = "black") +
  ggtitle("Histogram using ggplot2") +
xlab("Values") +
  ylab("Frequency")
# Boxplot
ggplot(data.frame(data), aes(y = data)) +
  geom_boxplot(fill = "orange", color = "black") +
  ggtitle("Boxplot using ggplot2") +
ylab("Values")

Program 4: Missing Data Handling


# R Program: Missing Data Handling Demonstration in Data Preprocessing
# -----------------------------------------------------------------------------
# Load necessary libraries
# The 'dplyr' and 'tidyr' packages are part of the 'tidyverse' and are
# excellent for data manipulation and cleaning.
if (!require(tidyverse)) {
install.packages("tidyverse", dependencies = TRUE)
library(tidyverse)
}
# -----------------------------------------------------------------------------
# Step 1: Create a Sample Dataset with Missing Values
# -----------------------------------------------------------------------------
cat("Step 1: Creating a sample dataset with missing values...\n\n")
set.seed(42)
customer_data <- tibble(
customer_id = 1:10,
age = c(25, 30, NA, 45, 28, 50, NA, 35, 60, 42),
annual_income = c(50000, 65000, 48000, NA, 75000, 90000, 55000, 62000, NA, 80000),
gender = c("Male", "Female", "Male", NA, "Female", "Male", "Female", "Male", NA, "Female"),
satisfaction_score = c(4, 5, 3, 2, 5, 4, 3, NA, 5, 4)
)
print(customer_data)
# -----------------------------------------------------------------------------
# Step 2: Identify Missing Values
# -----------------------------------------------------------------------------
cat("\n\nStep 2: Identifying missing values...\n\n")
cat("Output of is.na() on the 'age' column:\n")
print(is.na(customer_data$age))
cat("\nNumber of missing values per column:\n")
print(colSums(is.na(customer_data)))
cat("\nAre there any missing values in the entire dataset?\n")
print(any(is.na(customer_data)))
# -----------------------------------------------------------------------------
# Step 3: Common Methods for Handling Missing Values
# -----------------------------------------------------------------------------
cat("\n\nMethod A: Deleting rows with missing values (Listwise Deletion)...\n\n")
data_no_na <- na.omit(customer_data)
cat("Original data dimensions:", dim(customer_data), "\n")
cat("Dimensions after deleting rows:", dim(data_no_na), "\n")
print(data_no_na)
cat("\n\nMethod B: Imputing 'annual_income' with its median...\n\n")
data_imputed_median <- customer_data
median_income <- median(data_imputed_median$annual_income, na.rm = TRUE)
data_imputed_median$annual_income[is.na(data_imputed_median$annual_income)] <-
median_income
cat("Median annual income:", median_income, "\n")
cat("Data after median imputation:\n")
print(data_imputed_median)
cat("\n\nMethod C: Imputing 'gender' with its mode...\n\n")
data_imputed_mode <- customer_data
mode_gender <- names(which.max(table(data_imputed_mode$gender)))
data_imputed_mode$gender[is.na(data_imputed_mode$gender)] <- mode_gender
cat("Mode gender:", mode_gender, "\n")
cat("Data after mode imputation:\n")
print(data_imputed_mode)
cat("\n\nMethod D: Using `tidyr::fill()` for Time-Series or Sequential Data ...\n\n")
time_series_data <- tibble(
date = 1:5,
stock_price = c(100, NA, 105, NA, 110)
)
filled_data <- time_series_data %>%
fill(stock_price, .direction = "down")
cat("Original sequential data:\n")
print(time_series_data)
cat("\nData after filling 'down':\n")
print(filled_data)
# -----------------------------------------------------------------------------
# Step 4: Acknowledging Advanced Imputation Methods
# -----------------------------------------------------------------------------
cat("\n\nStep 4: Acknowledging advanced methods...\n\n")
cat("While the above methods are simple and effective for many cases, advanced\n")
cat("imputation techniques exist for more complex scenarios. These methods,\n")
cat("such as K-Nearest Neighbors (KNN) or Multiple Imputation by Chained\n")
cat("Equations (MICE), use the relationships between variables to estimate\n")
cat("missing values. They often provide more accurate and robust results\n")
cat("than simple mean/median imputation.\n")

Program 5: Data Normalization


# R Program: Data Normalization Demonstration
# -----------------------------------------------------------------------------
# Load necessary libraries
if (!require(tidyverse)) {
install.packages("tidyverse", dependencies = TRUE)
library(tidyverse)
}
# -----------------------------------------------------------------------------
# Step 1: Create a Sample Dataset
# -----------------------------------------------------------------------------
cat("Step 1: Creating a sample dataset with different scales...\n\n")
customer_data <- tibble(
customer_id = 1:10,
annual_income = c(50, 65, 48, 95, 75, 90, 55, 62, 120, 80),
satisfaction_score = c(4, 5, 3, 9, 7, 8, 5, 6, 10, 7)
)
print(customer_data)
cat("\nSummary of original data:\n")
print(summary(customer_data))
# -----------------------------------------------------------------------------
# Step 2: Min-Max Normalization (Scaling to a [0, 1] range)
# -----------------------------------------------------------------------------
cat("\n\nStep 2: Performing Min-Max Normalization...\n\n")
min_max_normalize <- function(x) {
return((x - min(x)) / (max(x) - min(x)))
}
normalized_min_max_data <- customer_data %>%
mutate(
income_min_max = min_max_normalize(annual_income),
score_min_max = min_max_normalize(satisfaction_score)
)
print(normalized_min_max_data)
cat("\nSummary of Min-Max normalized data:\n")
print(summary(select(normalized_min_max_data, starts_with("income_"), starts_with("score_"))))
# -----------------------------------------------------------------------------
# Step 3: Z-Score Standardization
# -----------------------------------------------------------------------------
cat("\n\nStep 3: Performing Z-Score Standardization...\n\n")
z_score_standardize <- function(x) {
return((x - mean(x)) / sd(x))
}
standardized_z_score_data <- customer_data %>%
mutate(
income_z_score = z_score_standardize(annual_income),
score_z_score = z_score_standardize(satisfaction_score)
)
print(standardized_z_score_data)
cat("\nSummary of Z-Score standardized data:\n")
print(summary(select(standardized_z_score_data, starts_with("income_"), starts_with("score_"))))
# -----------------------------------------------------------------------------
# Step 4: Visual Comparison (Optional but Recommended)
# -----------------------------------------------------------------------------
cat("\n\nStep 4: Creating a visual comparison plot...\n\n")
long_data <- standardized_z_score_data %>%
select(customer_id, annual_income, income_z_score, satisfaction_score, score_z_score) %>%
pivot_longer(
cols = -customer_id,
names_to = "variable",
values_to = "value"
)
normalization_plot <- ggplot(long_data, aes(x = value, fill = variable)) +
geom_density(alpha = 0.5) +
facet_wrap(~variable, scales = "free") +
labs(
title = "Comparison of Original vs. Z-Score Standardized Data",
x = "Value",
y = "Density"
) +
theme_minimal() +
theme(legend.position = "none")
print(normalization_plot)
